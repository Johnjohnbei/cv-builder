**1. Add `serializeCV(cvElement: HTMLElement)` function:**

This function extracts the CV HTML and all stylesheets from the live DOM. It reuses the existing pattern from lines 37-46 of pdfExport.ts:

```typescript
export function serializeCV(cvElement: HTMLElement): { html: string; styles: string } {
  // Collect all <link rel="stylesheet"> tags
  const styleLinks = Array.from(document.querySelectorAll('link[rel="stylesheet"]'))
    .map(link => link.outerHTML)
    .join('\n');

  // Collect all inline CSS rules from all stylesheets
  const inlineStyles = Array.from(document.styleSheets)
    .map(sheet => {
      try { return Array.from(sheet.cssRules).map(r => r.cssText).join('\n'); }
      catch { return ''; }
    })
    .join('\n');

  // Clone and normalize the CV element
  const clone = cvElement.cloneNode(true) as HTMLElement;
  clone.style.transform = 'none';
  clone.style.position = 'relative';
  clone.style.top = '0';
  clone.style.left = '0';
  clone.style.width = '210mm';
  clone.style.height = 'auto';
  clone.style.overflow = 'visible';
  clone.style.border = 'none';
  clone.style.boxShadow = 'none';
  clone.style.margin = '0';

  return {
    html: clone.outerHTML,
    styles: `${styleLinks}\n<style>${inlineStyles}</style>`,
  };
}
```

**2. Add `serverlessPDF()` async function:**

This is the main new export. It:
- Accepts `cvElement`, `designSettings`, and callbacks for validation and loading state
- Runs the DOM pre-check validation (per D-14)
- Calls `serializeCV()` to get HTML+styles
- POSTs to `/api/generate-pdf` with `{ html, styles, pageLimit }` (per D-05)
- On success: creates blob URL, triggers download via `<a>` click with filename `CV_[date].pdf` (per D-12)
- On failure: falls back to `renderPDF()` (per D-13)

```typescript
export interface ServerlessPDFOptions {
  expectedText?: string;
  onValidation?: (result: ValidationResult) => void;
  onLoadingChange?: (loading: boolean) => void;
  onFallback?: (reason: string) => void;
}

export async function serverlessPDF(
  cvElement: HTMLElement,
  designSettings: DesignSettings,
  options?: ServerlessPDFOptions,
): Promise<void> {
  const pageLimit = designSettings.pageLimit || 1;

  // 1. Run DOM pre-check validation (per D-14)
  if (options?.expectedText && options?.onValidation) {
    const renderedText = cvElement.innerText || cvElement.textContent || '';
    const result = validateCVTextExtractability(renderedText, options.expectedText);
    options.onValidation(result);
  }

  // 2. Notify loading start
  options?.onLoadingChange?.(true);

  try {
    // 3. Serialize CV DOM
    const { html, styles } = serializeCV(cvElement);

    // 4. POST to serverless function
    const response = await fetch('/api/generate-pdf', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ html, styles, pageLimit }),
    });

    if (!response.ok) {
      throw new Error(`Server returned ${response.status}`);
    }

    // 5. Download PDF via blob URL (per D-12)
    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    const date = new Date().toISOString().slice(0, 10).replace(/-/g, '');
    a.download = `CV_${date}.pdf`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  } catch (error) {
    // 6. Fallback to window.print() (per D-13)
    console.error('Serverless PDF generation failed:', error);
    options?.onFallback?.('La generation serveur a echoue. Utilisation de l\'impression navigateur.');
    renderPDF(cvElement, designSettings, {
      expectedText: options?.expectedText,
      onValidation: options?.onValidation,
    });
  } finally {
    options?.onLoadingChange?.(false);
  }
}
```
