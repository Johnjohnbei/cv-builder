    /**
     * Lowercase + strip French diacritics. Pure.
     */
    export function normalizeForMatch(s: string): string {
      return s
        .toLowerCase()
        .normalize('NFD')
        .replace(/[\u0300-\u036f]/g, '') // strip combining marks (accents)
        .replace(/\s+/g, ' ')
        .trim();
    }

    /**
     * Strip common FR/EN suffixes for simple stemming.
     * Preserves words < 4 chars to avoid over-stripping (e.g., "ios").
     */
    export function stripSimpleSuffixes(word: string): string {
      if (word.length < 4) return word;
      const suffixes = [
        'ement', 'ation', 'ations', 'tions', 'tion',
        'ements', 'eurs', 'euses', 'euse', 'eur',
        'ing', 'ings', 'ed', 'er', 'ers',
        'es', 'ees', 'ees',
        's', 'x',
      ];
      for (const suf of suffixes) {
        if (word.length - suf.length >= 3 && word.endsWith(suf)) {
          return word.slice(0, -suf.length);
        }
      }
      return word;
    }

    /**
     * Stem a normalized phrase token-by-token (≥ 4 chars only).
     */
    function stemPhrase(s: string): string {
      return s
        .split(/\s+/)
        .map(t => (t.length >= 4 ? stripSimpleSuffixes(t) : t))
        .join(' ');
    }

    /**
     * Fuzzy keyword match — handles plural, accents, FR/EN suffix stemming.
     * Returns true if the keyword (or a stem of it) appears in the text with
     * word boundaries. ALL tokens of multi-word keywords must match.
     */
    export function matchKeywordFuzzy(keyword: string, text: string): boolean {
      if (matchKeyword(keyword, text)) return true;

      const nk = normalizeForMatch(keyword);
      const nt = normalizeForMatch(text);
      if (nk.length === 0) return false;

      // Exact match on normalized (accent-stripped) versions
      if (matchKeyword(nk, nt)) return true;

      // Stem both sides
      const stemmedKeyword = stemPhrase(nk);
      const stemmedText = stemPhrase(nt);

      // Split keyword into tokens; all must appear in stemmed text with word boundaries
      const tokens = stemmedKeyword.split(/\s+/).filter(t => t.length >= 3);
      if (tokens.length === 0) return false;

      for (const tok of tokens) {
        if (!matchKeyword(tok, stemmedText)) return false;
      }
      return true;
    }

    import {
      matchKeywordFuzzy,
      normalizeForMatch,
      stripSimpleSuffixes,

    import type { CVData } from '@/src/shared/types';

    describe('normalizeForMatch', () => {
      it('lowercases and strips French accents', () => {
        expect(normalizeForMatch('Développeur Créatif')).toBe('developpeur creatif');
      });
      it('collapses whitespace', () => {
        expect(normalizeForMatch('  hello    world  ')).toBe('hello world');
      });
    });

    describe('stripSimpleSuffixes', () => {
      it('strips FR plural -s', () => {
        expect(stripSimpleSuffixes('designers')).toBe('design');
      });
      it('strips -tion', () => {
        expect(stripSimpleSuffixes('gestion')).toBe('ges');
      });
      it('strips -ing', () => {
        expect(stripSimpleSuffixes('coding')).toBe('cod');
      });
      it('protects words < 4 chars', () => {
        expect(stripSimpleSuffixes('iOS')).toBe('iOS');
        expect(stripSimpleSuffixes('UI')).toBe('UI');
      });
      it('does not over-strip short roots', () => {
        // "les" would become "l" without protection — must stay "les"
        expect(stripSimpleSuffixes('les')).toBe('les');
      });
    });

    describe('matchKeywordFuzzy', () => {
      it('matches exact keyword (fast path)', () => {
        expect(matchKeywordFuzzy('React', 'I use React daily')).toBe(true);
      });
      it('matches plural in text', () => {
        expect(matchKeywordFuzzy('designer', 'team of 5 designers')).toBe(true);

        expect(matchKeywordFuzzy('user', 'worked with users and stakeholders')).toBe(true);
      });

        expect(matchKeywordFuzzy('développeur', 'developpeur senior')).toBe(true);
