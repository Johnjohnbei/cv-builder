    3. In `EditorPage.tsx`, add the ATS mode toggle handler. This goes after the existing state declarations (around line 53). Add state to track the pre-ATS template:
       ```typescript
       const [preAtsTemplate, setPreAtsTemplate] = useState<string | null>(null);
       ```

    4. Create a handler function `handleAtsModeChange` in EditorPage:
       ```typescript
       const handleAtsModeChange = (enabled: boolean) => {
         if (enabled) {
           // Check if current template is ATS-compatible
           const compat = TEMPLATE_ATS_COMPAT[selectedTemplate];
           if (compat === 'limited') {
             // Save current template before auto-switching (per D-15)
             setPreAtsTemplate(selectedTemplate);
             // Auto-switch to fallback (per D-07)
             setSelectedTemplate(ATS_FALLBACK_TEMPLATE);
             setDesignSettings(prev => ({
               ...prev,
               atsMode: true,
               template: ATS_FALLBACK_TEMPLATE,
             }));
             notify({ message: `Template incompatible ATS — basculé vers Classic`, type: 'success' });
           } else {
             setDesignSettings(prev => ({ ...prev, atsMode: true }));
           }
         } else {
           // Deactivating ATS mode (per D-14, D-15)
           if (preAtsTemplate) {
             // Restore original template
             setSelectedTemplate(preAtsTemplate);
             setDesignSettings(prev => ({
               ...prev,
               atsMode: false,
               template: preAtsTemplate,
             }));
             setPreAtsTemplate(null);
           } else {
             setDesignSettings(prev => ({ ...prev, atsMode: false }));
           }
           notify({ message: 'Mode ATS désactivé — le CV ne sera plus optimisé ATS', type: 'error' });
         }
       };
       ```

    5. Add the required imports in EditorPage.tsx:
       ```typescript
       import { TEMPLATE_ATS_COMPAT, ATS_FALLBACK_TEMPLATE } from '@/src/features/editor/lib/atsRules';
       ```

    6. Pass the new props to `<EditorHeader>`:
       ```tsx
       atsMode={designSettings.atsMode ?? false}
       onAtsModeChange={handleAtsModeChange}
       ```

    Per D-04: atsMode does NOT mutate other DesignSettings fields — it is a rendering overlay. The only mutation is the boolean itself (and template on auto-switch).
